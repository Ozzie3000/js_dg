Property Access expressions

  • var b = {x:1, y:{z:3}}; //object
  • var a = [b,4,[5,6]]; //array that contains the object
  • b.x //= 1 ; //expression o with property x
  • b.y.z // = 3 ; //property z of expression x.o
  • b["x"] // = 1 ; //property x of object o
  • a[1] // = 4 ; //element at index 1 of expression a
  • a[2]["1"] // = 6 ; //element at index 1 of expression a[2]
  • a[0].x // = 1 //property x of expression a[0]
